 works  
 
  select s.sessionTitle, s.speakerName, s.sessionDate, r.roomName
				from attendee a
				INNER JOIN (registration reg
					INNER JOIN (room r
						INNER JOIN session s
							on r.roomID = s.roomID)
						on s.sessionID = reg.sessionID)
					on a.attendeeID = reg.attendeeID;
